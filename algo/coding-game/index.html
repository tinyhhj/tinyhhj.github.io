CodingGame

코딩게임 Spring Challenge를 우연히 접하고 참가해보았다.
기존의 한가지 문제만을 해결하는 단순 문제와는 다르게 복합적인 문제를 해결하는 문제로 더 어렵게 느껴졌다.

코딩게임은 주어진 api를 이용해서, 게임 로직을 구현한뒤 다른사람이 작성한 봇과 승부해 랭킹을 가리는 형식인데,

아무래도 기존 문제와는 다르게, 그래픽도 있고, 다른사람과 순위로 경쟁을 한다는 점에서 더 열중하게되고, 재밌었다.

처음에 제공되는 스타터 킷을 이용해서 코드를 작성하였는데, 이번 경기를 참가하면서, 게임 내 자료구조가 고정되 있다는 점에서, 절차지향적 프로그래밍이 훨씬 편하다고 느껴졌다.

게임 내 공통적으로 사용하게 되는 자료는 Cell과 Tree이었고, 이 두 자료구조를 잘 조합해 메소드를 만드는 방식으로 진행했었고,최대한 에러가 발생하지 않도록, 검증로직을 꼼꼼하게 체크해줬다.

문제의 경우 해법이 명확하게 보이지 않아서, 각 Action별로 카테고리를 나눠, 카테고리 별 우선순위를 정해 명령을 수행하는 방식으로 진행했다.

턴마다 가능한 액션들을 카테고리별로 모으고, 이 중 원하는 액션들을 한차례 필터링해준다(ex: SEED의 경우 origin 위치에서 2만큼 떨어진 곳). 필터링 결과를 특정 조건에 맞춰 정렬한 뒤, 가장 우선순위가 높은 액션을 수행하는 식으로 구성했다.

의도하지는 않았지만, 액션의 가능유무를 체크하는 isValidAction 부분과, 필터링된 액션들의 정렬을 결정하는 calculateValue, 액션을 최종 선택하는 nextAction으로 크게 책임을 분리해놓으니, 문제가 생겼을 때 변경점이 분리되는 장점을 느낄 수 있었다.

비록 골드리그 200-300위 정도에 그쳤지만, 앞으로 꾸준히 참가해서 복합적인 사고능력을 늘려야겠다.

class Cell:
    def __init__(self, idx,richness,neighbors):
        self.idx = int(idx)
        self.richness = richness
        self.neighbors = neighbors
    def __str__(self):
        return f'{self.idx}:{self.richness}:{self.neighbors}'
    def isUsable(self):
        return self.richness > 0

class Tree:
    def __init__(self, idx,size,is_mine,is_dormant):
        self.idx = int(idx)
        self.size = size
        self.is_mine = is_mine
        self.is_dormant = is_dormant
    def isTree(self):
        return self.size > 0
    def isDormant(self):
        return self.is_dormant
    def isActive(self):
        return not self.is_dormant
    
class ActionType(Enum):
    WAIT = 'WAIT'
    SEED = 'SEED'
    GROW = 'GROW'
    COMPLETE = 'COMPLETE'
class Action:
    GROW_COSTS=[0,1,3,7]
    MAX_COST = 9999
    MAX_TREE_CNT = [1,2,3,5] # 까지 가능
    def __init__(self, type, target_idx = None, origin_idx= None):
        self.type = type
        self.target_idx = target_idx
        self.origin_idx = origin_idx
        self.cost= None
    def __str__(self):
        if self.type == ActionType.WAIT:
            return 'WAIT'
        elif self.type == ActionType.SEED:
            return f'SEED {self.origin_idx} {self.target_idx}'
        else:
            return f'{self.type.name} {self.target_idx}'
        
    @staticmethod
    def parse(action_string):
        split = action_string.split(' ')
        if split[0] == ActionType.WAIT.name:
            return Action(ActionType.WAIT)
        elif split[0] == ActionType.SEED.name:
            return Action(ActionType.SEED, int(split[2]), int(split[1]))
        elif split[0] == ActionType.GROW.name:
            return Action(ActionType.GROW, int(split[1]))
        else:
            return Action(ActionType.COMPLETE, int(split[1]))
    def caculateSunPower(self, game):
        if self.type == ActionType.WAIT:
            return 0
        elif self.type == ActionType.SEED:
            cost = len(game.getMyTrees([0]))
            return cost if cost < Action.MAX_TREE_CNT[0] else Action.MAX_COST
        elif self.type == ActionType.GROW:
            tree = game.getTreeByIdx(self.target_idx)
            ownCost = len(game.getMyTrees([tree.size+1]))
            growCost = Action.GROW_COSTS[tree.size+1]
            if game.day < 21:
                return Action.MAX_COST if ownCost >= Action.MAX_TREE_CNT[tree.size+1] else ownCost + growCost
            else:
                if game.day == 22 and tree and tree.size == 2 and ownCost + growCost + 4 < 3 * Util.getExpectPoint(game,tree.idx):
                    return  ownCost + growCost + 4
                return Action.MAX_COST
        elif self.type == ActionType.COMPLETE:
            completeTrees = len(game.getMyTrees([3]))
            childTrees = len(game.getMyTrees([2]))
            # print(f'complete {completeTrees} {Action.MAX_TREE_CNT[3]} {childTrees} {game.harvest}',file=sys.stderr,flush=True)
            if completeTrees >= Action.MAX_TREE_CNT[3]:
                return 4
            return Action.MAX_COST
    def isValidAction(self,game):
        if not self.cost is None:
            return self.cost
        if self.type == ActionType.WAIT:
            self.cost = 0
            return self.cost
        elif self.type == ActionType.SEED:
            if Util.isValidIndex(self.origin_idx) and Util.isValidIndex(self.target_idx): 
                tree = game.getTreeByIdx(self.origin_idx)
                cell = game.board[self.target_idx]
                targetTree = game.getTreeByIdx(self.target_idx)
                effectIdxes = set()
                for tidx in game.getTreeIdxes(game.getMyTrees([3])):
                    for effectIdx in Util.getEffectIdxes(game,tidx):
                        effectIdxes.add(effectIdx)
                # print('Debug message..',self.type.name,self.origin_idx,self.target_idx,tree,targetTree,file=sys.stderr,flush=True)
                if tree and tree.isActive() and cell.isUsable() and not targetTree:
                    cost = self.caculateSunPower(game)
                    if game.my_sun >= cost and not self.target_idx in effectIdxes:
                        self.cost = cost
                        return self.cost
        elif self.type == ActionType.GROW:
            if Util.isValidIndex(self.target_idx): 
                tree = game.getTreeByIdx(self.target_idx)
                if tree and tree.size < 3 and tree.isActive():
                    cost = self.caculateSunPower(game)
                    if game.my_sun >= cost:
                        self.cost = cost
                        return self.cost
        elif self.type == ActionType.COMPLETE:
            if Util.isValidIndex(self.target_idx): 
                tree = game.getTreeByIdx(self.target_idx)
                if tree and tree.size == 3 and tree.isActive():
                    cost = self.caculateSunPower(game)
                    if game.my_sun >= cost:
                        self.cost = cost
                        return self.cost
        self.cost = Action.MAX_COST
        return self.cost
        
class Util:
    countW = min(0.75,random.random())
    richnessW = 1-countW
    @staticmethod
    def isValidIndex(idx):
        return 0 <= idx <= 37
    @staticmethod
    def calculateValue(game, action):
        if action.type == ActionType.SEED:
            # 주변에 많을수록 좋다, richness가 높을수록 좋다
            # -3-2-1 321
            opTreesIdxes = game.getTreeIdxes(game.getOpTrees())
            print('seed',action.origin_idx,action.target_idx, -(Util.countW * -Util.getOpTreeCount(game,action.target_idx)[1]+ Util.richnessW * game.board[action.target_idx].richness),file=sys.stderr,flush=True)
            if game.day < 17:
                return (-Util.getMyTreeCount(game,action.target_idx)[1],-game.board[action.target_idx].richness)
            else:
                return (-game.board[action.target_idx].richness,-Util.getMyTreeCount(game,action.target_idx)[1])
        elif action.type == ActionType.GROW:
            tree = game.getMyTreeByIdx(action.target_idx)
            print('grow', action.target_idx,(-tree.size,*Util.getOpTreeCount(game,action.target_idx), action.isValidAction(game)),file=sys.stderr,flush=True)
            return (-tree.size,*Util.getOpTreeCount(game,action.target_idx), action.isValidAction(game))
        elif action.type == ActionType.COMPLETE:
            # tree = game.getTreeByIdx(action.target_idx)
            urgent, cnt = Util.getOpTreeCount(game,action.target_idx)
            print('complete', action.target_idx,(-cnt, -urgent),file=sys.stderr,flush=True)
            return (-game.board[action.target_idx].richness, -cnt, -urgent)
        
    @staticmethod
    def getPositionsWithConditionTrue(game,st,condition):
        res = []
        visited = [st]
        q = [(st,0)]
        while q:
            idx,dis = q.pop()
            for nidx in game.board[idx].neighbors:
                if condition(nidx,dis+1) and not nidx in visited:
                    visited.append(idx)
                    res.append(nidx)
                    q.append((nidx,dis+1))
        return res

                trees[i] = tree

class BuildManager:

    def getPossibleSeedActions(self):
        board = self.game.board

        q = []

        for t in self.game.getMyTrees():

            if t.isTree() and t.isActive():
                visited = []
                q.append((t.idx,0))
                visited.append(t.idx)
                while q:
                    idx,dis = q.pop(0)
                    for nidx in board[idx].neighbors:
                        # size 내에 다 검색하는데 비어있는건 중복이 안되야하지만 트리가 있거나 unusable은 탐색은 해야한다.
                        if  dis+1 <= t.size and not nidx in visited and Util.isValidIndex(nidx):
                            visited.append(nidx)
                            q.append((nidx,dis+1))
                            if board[nidx].isUsable() and self.game.getTreeByIdx(nidx) is None and not nidx in possibleIdxes and dis >= 1:
                                possibleIdxes.add(nidx)
                                actions.append(Action.parse(f'SEED {t.idx} {nidx}'))
        # for action in actions:
        #     print('before validate action', action.origin_idx, action.target_idx, Util.calculateValue(self.game,action),file=sys.stderr, flush=True)
        validActions = list(filter(lambda action: action.isValidAction(self.game) != Action.MAX_COST, actions))

        if validActions:
            # for action in validActions:
            #     print('seed val', action.origin_idx, action.target_idx, Util.calculateValue(self.game,action),file=sys.stderr, flush=True)
            return sorted(validActions,key=lambda action:Util.calculateValue(self.game,action))[0]
        else:
            return []
    def getPossibleGrowActions(self):
        actions = []
        for tree in self.game.getMyTrees([0,1,2]):
            actions.append(Action(ActionType.GROW,tree.idx))
        validActions = list(filter(lambda action: action.isValidAction(self.game) != Action.MAX_COST,actions))
        if validActions:
            return sorted(validActions,key=lambda action:Util.calculateValue(self.game,action))[0]
        else:
            return []
    def getPossibleCompleteActions(self):
        actions = []
        for tree in self.game.getMyTrees([3]):
            actions.append(Action(ActionType.COMPLETE, tree.idx))
        validActions = list(filter(lambda action: action.isValidAction(self.game) != Action.MAX_COST,actions))
        if validActions:
            return sorted(validActions,key=lambda action:Util.calculateValue(self.game,action))[0]
        else:
            return []

class Game:
    def __init__(self):
        self.day = 0
        self.nutrients = 0
        self.board = []
        self.trees = []
        self.possible_actions = []
        self.my_sun = 0
        self.my_score = 0

        self.opponent_is_waiting = 0

    def refresh(self,day):
        if day > self.day:
            self.harvest = False
    def compute_next_action(self):
        return self.possible_actions[0][1]
    def getMyTrees(self, size = [0,1,2,3]):
        return list(filter(lambda t:t.is_mine and t.size in size, self.trees))
    def getOpTrees(self,size = [0,1,2,3]):
        return list(filter(lambda t:not t.is_mine and t.size in size, self.trees))
    def getTreeIdxes(self,trees= None):
        return list(map(lambda t: t.idx if t else -1,self.trees if trees is None else trees))

        res = list(filter(lambda t:t.is_mine and idx == t.idx, self.trees))
